磁盘盘符绑定

2019-08-21 1 min read # Linux

脚本：

#!/bin/bash
disk="a b c d e f g h i j k l i j k m n"
for i in disk;doa=‘/usr/sbin/udevadminfo−qpath−n/dev/sd{disk}; do a=`/usr/sbin/udevadm info -q path -n /dev/sddisk;doa=‘/usr/sbin/udevadminfo−qpath−n/dev/sd{i}`;
if [ ! -n "a" ]; then break 1 ; else echo DEVPATH=="\"{a}"", NAME=""sd${i}"", MODE=""0660"">>/etc/udev/rules.d/10-mydisk.rules;
fi
done
