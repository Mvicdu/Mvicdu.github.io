将表内容转储到文件

2020-03-28

SET HEADING OFF ECHO OFF FEEDBACK OFF
SET PAGESIZE 0

SELECT
'INSERT INTO departments_test VALUES
(' || department_id ||','''|| department_name ||''',' || location_id || ');'
as "Insert Statements Script"
FROM departments
/

SET PAGESIZE 24
SET HEADIN G ON ECHO OFF FEEDBACK ON
